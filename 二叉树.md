```java
  while(root!=null){

  st.push(root)

  }
  //中序遍历
  root=st.pop();
  res.add
  root=root.right;
  //后序遍历
  TreeNode node=st.peek();
  if(node.right!=null&&node.right!=pre)
  root=node.right
  else{
  root=st.pop()
  res.add
  pre=root
  root=null
  }
}
```
